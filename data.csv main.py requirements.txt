import pandas as pd
from sklearn.linear_model import LinearRegression

data = {
    'students': [120, 150, 130, 160, 180],
    'food_demand': [100, 130, 110, 150, 170]
}

df = pd.DataFrame(data)

X = df[['students']]
y = df['food_demand']

model = LinearRegression()
model.fit(X, y)

print("Prediction for 140 students:", model.predict([[140]]))
